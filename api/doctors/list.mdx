---
title: 'List Doctors'
api: 'GET /api/doctors'
description: 'Retrieve a list of all doctors with optional filtering'
---

## Query Parameters

<ParamField query="specialty" type="string">
  Filter doctors by medical specialty (e.g., "Cardiology", "Pediatrics", "Dermatology")
</ParamField>

<ParamField query="availability" type="boolean">
  Filter doctors by availability status. Set to `true` to show only available doctors
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of results per page (max: 100)
</ParamField>

<ParamField query="search" type="string">
  Search doctors by name, specialty, or license number
</ParamField>

## Response

<ResponseField name="doctors" type="array">
  Array of doctor objects
  
  <ResponseField name="id" type="string">
    Unique identifier for the doctor
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Full name of the doctor
  </ResponseField>
  
  <ResponseField name="specialty" type="string">
    Medical specialty or specialization
  </ResponseField>
  
  <ResponseField name="license_number" type="string">
    Professional medical license number
  </ResponseField>
  
  <ResponseField name="email" type="string">
    Doctor's contact email address
  </ResponseField>
  
  <ResponseField name="phone" type="string">
    Doctor's contact phone number
  </ResponseField>
  
  <ResponseField name="bio" type="string">
    Professional biography and credentials
  </ResponseField>
  
  <ResponseField name="available" type="boolean">
    Whether the doctor is currently accepting new patients
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp of when the doctor was added to the system
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    ISO 8601 timestamp of last update
  </ResponseField>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata
  
  <ResponseField name="page" type="integer">
    Current page number
  </ResponseField>
  
  <ResponseField name="limit" type="integer">
    Results per page
  </ResponseField>
  
  <ResponseField name="total" type="integer">
    Total number of doctors matching the query
  </ResponseField>
  
  <ResponseField name="pages" type="integer">
    Total number of pages
  </ResponseField>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.saludya.com/api/doctors?specialty=Cardiology&availability=true&page=1&limit=10" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```bash Filter by Search
curl -X GET "https://api.saludya.com/api/doctors?search=Garcia" \
  -H "Authorization: Bearer YOUR_API_KEY"
```
</RequestExample>
