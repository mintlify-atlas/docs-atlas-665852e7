---
title: 'Create Appointment'
api: 'POST /api/appointments'
description: 'Schedule a new medical appointment'
---

Create a new appointment in the system. The appointment will be created with a `scheduled` status by default.

## Request Body

<ParamField body="patient_id" type="string" required>
  ID of the patient for this appointment
</ParamField>

<ParamField body="doctor_id" type="string" required>
  ID of the doctor who will attend the appointment
</ParamField>

<ParamField body="date" type="string" required>
  Date of the appointment in ISO 8601 format (YYYY-MM-DD)
</ParamField>

<ParamField body="time" type="string" required>
  Time of the appointment in 24-hour format (HH:MM)
</ParamField>

<ParamField body="duration_minutes" type="integer" default="30">
  Expected duration of the appointment in minutes
</ParamField>

<ParamField body="reason" type="string" required>
  Reason for the appointment visit
</ParamField>

<ParamField body="notes" type="string">
  Additional notes or special instructions for the appointment
</ParamField>

<ParamField body="send_notification" type="boolean" default="true">
  Whether to send confirmation notification to the patient
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique appointment identifier
</ResponseField>

<ResponseField name="patient_id" type="string">
  ID of the patient
</ResponseField>

<ResponseField name="patient_name" type="string">
  Full name of the patient
</ResponseField>

<ResponseField name="doctor_id" type="string">
  ID of the assigned doctor
</ResponseField>

<ResponseField name="doctor_name" type="string">
  Full name of the doctor
</ResponseField>

<ResponseField name="specialty" type="string">
  Medical specialty for the appointment
</ResponseField>

<ResponseField name="date" type="string">
  Appointment date (ISO 8601 format)
</ResponseField>

<ResponseField name="time" type="string">
  Appointment time (HH:MM format)
</ResponseField>

<ResponseField name="duration_minutes" type="integer">
  Expected duration of the appointment in minutes
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the appointment (will be `scheduled` for new appointments)
</ResponseField>

<ResponseField name="reason" type="string">
  Reason for the appointment visit
</ResponseField>

<ResponseField name="notes" type="string">
  Additional notes or special instructions
</ResponseField>

<ResponseField name="created_at" type="string">
  Timestamp when the appointment was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  Timestamp when the appointment was last updated
</ResponseField>

## Example Request

```bash
curl -X POST "https://api.saludya.com/api/appointments" \
  -H "Authorization: Bearer YOUR_API_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "patient_id": "pat_5x8y2z4a6b",
    "doctor_id": "doc_3j5k7m9n1p",
    "date": "2026-03-15",
    "time": "14:30",
    "duration_minutes": 30,
    "reason": "Consulta de seguimiento",
    "notes": "Paciente reporta dolor en el pecho",
    "send_notification": true
  }'
```

```json
  "error": {

    "details": {

```
