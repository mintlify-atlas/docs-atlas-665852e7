---
title: 'List Appointments'
api: 'GET /api/appointments'
description: 'Retrieve a paginated list of appointments with optional filtering and sorting'
---

Retrieve all appointments in the system with support for pagination, filtering by status, date range, patient, and doctor, as well as sorting options.

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of appointments per page (max: 100)
</ParamField>

<ParamField query="status" type="string">
  Filter by appointment status. Options: `scheduled`, `confirmed`, `in_progress`, `completed`, `cancelled`, `no_show`
</ParamField>

<ParamField query="patient_id" type="string">
  Filter appointments by patient ID
</ParamField>

<ParamField query="doctor_id" type="string">
  Filter appointments by doctor ID
</ParamField>

<ParamField query="date_from" type="string">
  Filter appointments from this date (ISO 8601 format: YYYY-MM-DD)
</ParamField>

<ParamField query="date_to" type="string">
  Filter appointments until this date (ISO 8601 format: YYYY-MM-DD)
</ParamField>

<ParamField query="sort_by" type="string" default="date">
  Field to sort by. Options: `date`, `created_at`, `status`
</ParamField>

<ParamField query="sort_order" type="string" default="asc">
  Sort order. Options: `asc`, `desc`
</ParamField>

## Response

<ResponseField name="appointments" type="array">
  Array of appointment objects
  
  <ResponseField name="id" type="string">
    Unique appointment identifier
  </ResponseField>
  
  <ResponseField name="patient_id" type="string">
    ID of the patient
  </ResponseField>
  
  <ResponseField name="patient_name" type="string">
    Full name of the patient
  </ResponseField>
  
  <ResponseField name="doctor_id" type="string">
    ID of the assigned doctor
  </ResponseField>
  
  <ResponseField name="doctor_name" type="string">
    Full name of the doctor
  </ResponseField>
  
  <ResponseField name="specialty" type="string">
    Medical specialty for the appointment
  </ResponseField>
  
  <ResponseField name="date" type="string">
    Appointment date (ISO 8601 format)
  </ResponseField>
  
  <ResponseField name="time" type="string">
    Appointment time (HH:MM format)
  </ResponseField>
  
  <ResponseField name="duration_minutes" type="integer">
    Expected duration of the appointment in minutes
  </ResponseField>
  
  <ResponseField name="status" type="string">
    Current status of the appointment: `scheduled`, `confirmed`, `in_progress`, `completed`, `cancelled`, `no_show`
  </ResponseField>
  
  <ResponseField name="reason" type="string">
    Reason for the appointment visit
  </ResponseField>
  
  <ResponseField name="notes" type="string">
    Additional notes or special instructions
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    Timestamp when the appointment was created
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    Timestamp when the appointment was last updated
  </ResponseField>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata
  
  <ResponseField name="current_page" type="integer">
    Current page number
  </ResponseField>
  
  <ResponseField name="total_pages" type="integer">
    Total number of pages
  </ResponseField>
  
  <ResponseField name="total_count" type="integer">
    Total number of appointments matching the filters
  </ResponseField>
  
  <ResponseField name="per_page" type="integer">
    Number of items per page
  </ResponseField>
</ResponseField>

## Example Request

```bash
curl -X GET "https://api.saludya.com/api/appointments?page=1&limit=10&status=scheduled&date_from=2026-03-06&sort_by=date&sort_order=asc" \
  -H "Authorization: Bearer YOUR_API_TOKEN" \
  -H "Content-Type: application/json"
```
