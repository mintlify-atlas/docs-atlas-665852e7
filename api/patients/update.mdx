---
title: 'Update Patient'
api: 'PUT /api/patients/:id'
description: 'Update an existing patient record'
---

<Warning>
  This endpoint modifies Protected Health Information (PHI). All changes are audited and logged for HIPAA compliance.
</Warning>

## Path Parameters

<ParamField path="id" type="string" required>
  Unique patient identifier (e.g., `pat_1a2b3c4d5e6f`)
</ParamField>

## Request Body

<Info>
  All fields are optional. Only include fields you want to update.
</Info>

<ParamField body="name" type="string">
  Patient's full name
</ParamField>

<ParamField body="email" type="string">
  Patient's email address (must be unique)
</ParamField>

<ParamField body="phone" type="string">
  Patient's phone number in E.164 format
</ParamField>

<ParamField body="date_of_birth" type="string">
  Patient's date of birth in ISO 8601 format (YYYY-MM-DD)
</ParamField>

<ParamField body="medical_history" type="object">
  Patient's medical history information
  
  <ParamField body="conditions" type="array">
    List of existing medical conditions
  </ParamField>
  
  <ParamField body="allergies" type="array">
    List of known allergies
  </ParamField>
  
  <ParamField body="medications" type="array">
    List of current medications
  </ParamField>
  
  <ParamField body="blood_type" type="string">
    Blood type (A+, A-, B+, B-, AB+, AB-, O+, O-)
  </ParamField>
</ParamField>

<ParamField body="insurance" type="object">
  Insurance information
  
  <ParamField body="provider" type="string">
    Insurance provider name
  </ParamField>
  
  <ParamField body="policy_number" type="string">
    Insurance policy number
  </ParamField>
  
  <ParamField body="group_number" type="string">
    Insurance group number
  </ParamField>
  
  <ParamField body="expiration_date" type="string">
    Insurance expiration date (YYYY-MM-DD)
  </ParamField>
</ParamField>

<ParamField body="emergency_contact" type="object">
  Emergency contact information
  
  <ParamField body="name" type="string">
    Contact person's name
  </ParamField>
  
  <ParamField body="phone" type="string">
    Contact person's phone number
  </ParamField>
  
  <ParamField body="relationship" type="string">
    Relationship to patient
  </ParamField>
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique patient identifier
</ResponseField>

<ResponseField name="name" type="string">
  Patient's full name
</ResponseField>

<ResponseField name="email" type="string">
  Patient's email address
</ResponseField>

<ResponseField name="phone" type="string">
  Patient's phone number
</ResponseField>

<ResponseField name="date_of_birth" type="string">
  Patient's date of birth
</ResponseField>

<ResponseField name="medical_history" type="object">
  Patient's medical history
</ResponseField>

<ResponseField name="insurance" type="object">
  Insurance information
</ResponseField>

<ResponseField name="emergency_contact" type="object">
  Emergency contact information
</ResponseField>

<ResponseField name="created_at" type="string">
  Timestamp when patient was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  Timestamp when patient was last updated
</ResponseField>

<RequestExample>
```bash cURL - Update Phone and Email
curl -X PUT "https://api.saludya.com/api/patients/pat_1a2b3c4d5e6f" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "email": "maria.garcia.new@email.com",
    "phone": "+1-555-9999"
  }'
```

```bash Update Medical History
curl -X PUT "https://api.saludya.com/api/patients/pat_1a2b3c4d5e6f" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "medical_history": {
      "conditions": ["Hypertension", "Migraine", "Arthritis"],
      "allergies": ["Sulfa drugs", "Latex", "Shellfish"],
      "medications": ["Lisinopril 10mg", "Sumatriptan 50mg", "Ibuprofen 200mg"],
      "blood_type": "O+"
    }
  }'
```

```bash Update Insurance
curl -X PUT "https://api.saludya.com/api/patients/pat_1a2b3c4d5e6f" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "insurance": {
      "provider": "Cigna",
      "policy_number": "CIG789456123",
      "group_number": "GRP-3366",
      "expiration_date": "2026-12-31"
    }
  }'
```
</RequestExample>
