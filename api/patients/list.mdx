---
title: 'List Patients'
api: 'GET /api/patients'
description: 'Retrieve a paginated list of all patients in the system'
---

<Note>
  This endpoint returns sensitive medical data. Ensure proper authentication and authorization. All data is protected under HIPAA regulations.
</Note>

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of patients per page (max: 100)
</ParamField>

<ParamField query="search" type="string">
  Search patients by name, email, or phone number
</ParamField>

<ParamField query="sort" type="string" default="created_at">
  Sort field. Options: `name`, `email`, `created_at`, `date_of_birth`
</ParamField>

<ParamField query="order" type="string" default="desc">
  Sort order. Options: `asc`, `desc`
</ParamField>

<ParamField query="insurance" type="string">
  Filter by insurance provider
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of patient objects
  
  <ResponseField name="id" type="string">
    Unique patient identifier
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Patient's full name
  </ResponseField>
  
  <ResponseField name="email" type="string">
    Patient's email address
  </ResponseField>
  
  <ResponseField name="phone" type="string">
    Patient's phone number
  </ResponseField>
  
  <ResponseField name="date_of_birth" type="string">
    Patient's date of birth (ISO 8601 format)
  </ResponseField>
  
  <ResponseField name="insurance" type="object">
    Insurance information
    
    <ResponseField name="provider" type="string">
      Insurance provider name
    </ResponseField>
    
    <ResponseField name="policy_number" type="string">
      Insurance policy number
    </ResponseField>
    
    <ResponseField name="group_number" type="string">
      Insurance group number
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    Timestamp when patient was created
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    Timestamp when patient was last updated
  </ResponseField>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata
  
  <ResponseField name="current_page" type="integer">
    Current page number
  </ResponseField>
  
  <ResponseField name="per_page" type="integer">
    Number of items per page
  </ResponseField>
  
  <ResponseField name="total" type="integer">
    Total number of patients
  </ResponseField>
  
  <ResponseField name="total_pages" type="integer">
    Total number of pages
  </ResponseField>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.saludya.com/api/patients?page=1&limit=20&sort=name&order=asc" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"
```

```bash Search Example
curl -X GET "https://api.saludya.com/api/patients?search=maria&insurance=BlueCross" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"
```
</RequestExample>
