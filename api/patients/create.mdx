---
title: 'Create Patient'
api: 'POST /api/patients'
description: 'Create a new patient record in the system'
---

<Warning>
  This endpoint creates records containing Protected Health Information (PHI). Ensure compliance with HIPAA regulations and proper data encryption.
</Warning>

## Request Body

<ParamField body="name" type="string" required>
  Patient's full name
</ParamField>

<ParamField body="email" type="string" required>
  Patient's email address (must be unique)
</ParamField>

<ParamField body="phone" type="string" required>
  Patient's phone number in E.164 format (e.g., +1-555-0123)
</ParamField>

<ParamField body="date_of_birth" type="string" required>
  Patient's date of birth in ISO 8601 format (YYYY-MM-DD)
</ParamField>

<ParamField body="medical_history" type="object">
  Patient's medical history information
  
  <ParamField body="conditions" type="array">
    List of existing medical conditions
  </ParamField>
  
  <ParamField body="allergies" type="array">
    List of known allergies
  </ParamField>
  
  <ParamField body="medications" type="array">
    List of current medications
  </ParamField>
  
  <ParamField body="blood_type" type="string">
    Blood type (A+, A-, B+, B-, AB+, AB-, O+, O-)
  </ParamField>
</ParamField>

<ParamField body="insurance" type="object">
  Insurance information
  
  <ParamField body="provider" type="string" required>
    Insurance provider name
  </ParamField>
  
  <ParamField body="policy_number" type="string" required>
    Insurance policy number
  </ParamField>
  
  <ParamField body="group_number" type="string">
    Insurance group number
  </ParamField>
  
  <ParamField body="expiration_date" type="string">
    Insurance expiration date (YYYY-MM-DD)
  </ParamField>
</ParamField>

<ParamField body="emergency_contact" type="object">
  Emergency contact information
  
  <ParamField body="name" type="string">
    Contact person's name
  </ParamField>
  
  <ParamField body="phone" type="string">
    Contact person's phone number
  </ParamField>
  
  <ParamField body="relationship" type="string">
    Relationship to patient
  </ParamField>
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique patient identifier
</ResponseField>

<ResponseField name="name" type="string">
  Patient's full name
</ResponseField>

<ResponseField name="email" type="string">
  Patient's email address
</ResponseField>

<ResponseField name="phone" type="string">
  Patient's phone number
</ResponseField>

<ResponseField name="date_of_birth" type="string">
  Patient's date of birth
</ResponseField>

<ResponseField name="medical_history" type="object">
  Patient's medical history
</ResponseField>

<ResponseField name="insurance" type="object">
  Insurance information
</ResponseField>

<ResponseField name="emergency_contact" type="object">
  Emergency contact information
</ResponseField>

<ResponseField name="created_at" type="string">
  Timestamp when patient was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  Timestamp when patient was last updated
</ResponseField>
