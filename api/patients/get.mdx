---
title: 'Get Patient'
api: 'GET /api/patients/:id'
description: 'Retrieve detailed information about a specific patient'
---

<Note>
  This endpoint returns Protected Health Information (PHI). Access is logged for HIPAA compliance and audit purposes.
</Note>

## Path Parameters

<ParamField path="id" type="string" required>
  Unique patient identifier (e.g., `pat_1a2b3c4d5e6f`)
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique patient identifier
</ResponseField>

<ResponseField name="name" type="string">
  Patient's full name
</ResponseField>

<ResponseField name="email" type="string">
  Patient's email address
</ResponseField>

<ResponseField name="phone" type="string">
  Patient's phone number
</ResponseField>

<ResponseField name="date_of_birth" type="string">
  Patient's date of birth (ISO 8601 format)
</ResponseField>

<ResponseField name="medical_history" type="object">
  Patient's medical history information
  
  <ResponseField name="conditions" type="array">
    List of existing medical conditions
  </ResponseField>
  
  <ResponseField name="allergies" type="array">
    List of known allergies
  </ResponseField>
  
  <ResponseField name="medications" type="array">
    List of current medications
  </ResponseField>
  
  <ResponseField name="blood_type" type="string">
    Blood type
  </ResponseField>
</ResponseField>

<ResponseField name="insurance" type="object">
  Insurance information
  
  <ResponseField name="provider" type="string">
    Insurance provider name
  </ResponseField>
  
  <ResponseField name="policy_number" type="string">
    Insurance policy number
  </ResponseField>
  
  <ResponseField name="group_number" type="string">
    Insurance group number
  </ResponseField>
  
  <ResponseField name="expiration_date" type="string">
    Insurance expiration date
  </ResponseField>
</ResponseField>

<ResponseField name="emergency_contact" type="object">
  Emergency contact information
  
  <ResponseField name="name" type="string">
    Contact person's name
  </ResponseField>
  
  <ResponseField name="phone" type="string">
    Contact person's phone number
  </ResponseField>
  
  <ResponseField name="relationship" type="string">
    Relationship to patient
  </ResponseField>
</ResponseField>

<ResponseField name="appointments" type="object">
  Appointment statistics
  
  <ResponseField name="total" type="integer">
    Total number of appointments
  </ResponseField>
  
  <ResponseField name="upcoming" type="integer">
    Number of upcoming appointments
  </ResponseField>
  
  <ResponseField name="completed" type="integer">
    Number of completed appointments
  </ResponseField>
</ResponseField>

<ResponseField name="created_at" type="string">
  Timestamp when patient was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  Timestamp when patient was last updated
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.saludya.com/api/patients/pat_1a2b3c4d5e6f" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"
```
</RequestExample>
